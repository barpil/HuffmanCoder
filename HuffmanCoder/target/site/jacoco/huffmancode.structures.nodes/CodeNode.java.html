package huffmancode.structures.nodes;

public class CodeNode {
    private char character;
    private String code;

    public CodeNode(char character, String code){
        this.character=character;
        this.code=code;
    }

    public char getCharacter() {
        return character;
    }

    public String getCode() {
        return code;
    }
}
